df['month_year'] = pd.to_datetime(df['event_date']).dt.to_period('M')
df1=(
    df
    .groupby(['month_year'])
    .agg(
        events=('count', 'sum')
    )
    .reset_index()
)

#https://jupyter-gmaps.readthedocs.io/en/latest/tutorial.html
date = df1.month_year.dt.strftime('%m/%Y')
event = df1.events.values

fig = plt.figure(figsize = (15, 10))
plt.bar(date, event)
plt.title("Events per month")
plt.xlabel("Dates in Month-Year format")
plt.ylabel("Number of Events")
plt.xticks(rotation=60)
plt.figure(figsize=(3,6))
plt.show()
